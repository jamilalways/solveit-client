client/
│
├── public/
│   └── index.html
│
├── src/
│   ├── assets/            # images, icons, logos
│   ├── components/        # reusable UI components

│   ├── routes/            # routing setup
│   ├── services/          # API calls
│   ├── context/           # global state (auth etc.)
│   ├── hooks/             # custom hooks
│   ├── utils/             # helper functions
│   ├── styles/            # global css / tailwind config
│   ├── App.jsx
│   └── main.jsx
│
└── package.json
